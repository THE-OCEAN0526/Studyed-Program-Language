# while 用法

## 找最大值

```
#include <stdio.h>
int main(){
	// 由大往小找 
	int max;
	printf("MAX = ");
	scanf("%d", &max);
	
	int number = max;
	while(number > 0 &&
	// 迴圈要多加一個條件，不加會做到負的，變未定義型別 
	!(number%3 == 2 && number%5 == 3 && number%7 == 2)){
		// 不符合條件就找下一個(--)，直到找到符合條件跳出迴圈 
		number--;
	}
	// 符合韓信點兵最小值是23，所以要加一個判斷
	// 如果使用者輸入的值小於23，此時number會是0，則輸出沒有找到QQ
	if (number > 0){
		printf("%d\n", number);
	}else{
		printf("沒有找到QQ");
	}
	return 0;
}

```
# for 用法

## 找最大值

### 方法1 從最小走到最大

```
#include <stdio.h>
int main(){

	int max;
	printf("MAX = ");
	scanf("%d",&max);
	
	int answer = 0;
	int number;
	// 用number去走，走到最大 
	for (number = 1; number <= max; number++){
		if(number%3 == 2 && number%5 == 3 && number%7 == 2){
			// 符和條件，把number的值給answer，直到走到最大 
			answer = number;
		}
	}
	// 最後顯示出來 
	if (answer != 0){
		printf("%d\n",answer);
	}
	return 0;
}

```

### 方法2 從最大走到最小

```
#include <stdio.h>
int main(){

	int max;
	printf("MAX = ");
	scanf("%d",&max);
	
	int answer = 0;
	int number;
	// 用number去走，走到最大 
	/* for迴圈多加answer == 0這個條件，
	如果不是0跳出迴圈，沒有這個條件不會找到最大之後就停**/ 
	for (number = max; number >= 1 && answer == 0; number--){
		if(number%3 == 2 && number%5 == 3 && number%7 == 2){
			// 符和條件，把number的值給answer，直到走到最小
			answer = number;
		}
	}
	// 最後顯示出來 
	if (answer != 0){
		printf("%d\n",answer);
	}
	return 0;
}
```
