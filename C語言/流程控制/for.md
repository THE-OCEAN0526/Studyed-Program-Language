# for 述句
- for (初始式; 條件式; 迴圈式) {程式片段}
- 執行「初始式」。
- 當「條件式」 成立時， 執行 「程式片段」。 否則就結束迴圈。
- 執行「迴圈式」 後在回到上一步。
```
for(初始式; 條件式; 迴圈式){       
  程式片段                       
}                                
  // 相當於
  
初始式;
while (條件式) {
  程式片段
  迴圈式;
}
```
```
int count = 1;
while(count <= 10) {
  printf("%\n", count);
  count = count + 1;
}
 // 相當於
 int count;
 for(count = 1; count <= 10; count++){
  printf("%d\n", count);
 }
```

#include <stdio.h>
#include <stdlib.h>
int main(){
	int x, y, N;
	while(scanf("%d",&N)==1){               
		for(x=1;x<=N;x++){						 
			for(y=1;y<=N;y++){					 
				if(y==1 || x==N || x==y){		 
					printf("*");				 
				}else{							 
					printf(" ");				 
				}								 
			}									
			printf("\n");
		}
	}

}
